    // by Max IBU
    $("#findMaxIBU").keyup(() => {
        let link;
        if ($("#findMaxIBU").val() == "") {
            link = "https://api.punkapi.com/v2/beers?ibu_lt";
        }
        else {
            link = "https://api.punkapi.com/v2/beers?per_page=80&beer_name=" + $("#findMaxIBU").val();
        }
        $('#card').empty();
        xhttp.open("GET", link, true);
        xhttp.send();
    })
